繪製台灣各鄉鎮的1st-order鄰居人口密度的面量圖

for (i in 1:length(Popn_TWN.nb)) {
  population <- 0
  area <- 0
  if (Popn_TWN.nb[[i]] == 0)
    Popn_TWN$NB_DENSITY[i] <- 0
  else {
    for (j in Popn_TWN.nb[[i]]) {
      population = population + Popn_TWN$TOTAL_CNT[j]
      area = area + Popn_TWN$AREA[j]
    }
    Popn_TWN$NB_DENSITY[i] <- 10^6 * population / area
  }
}

shades = auto.shading(Popn_TWN$NB_DENSITY, n = 5, cols = brewer.pal(5, "Greens"))
choropleth(Popn_TWN, Popn_TWN$NB_DENSITY, shades, xlim = c(170000, 180000), ylim = c(2440000, 2800000))
