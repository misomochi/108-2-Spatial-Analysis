Pollution_Map = function(arg1) {
  par(mfrow = c(1, 2))
  pr = qnorm(1 - arg1, mean = mean(PSI), sd = sd(PSI));
  
  par(family = "BiauKai");
  par(mar = c(0, 0, 1, 0));
  plot(Popn.TWN);
  plot(EPA.STN, add = TRUE, col = ifelse(PSI > pr, 'red', 'blue'), pch = 19);
  map.scale(63030.22, 2472112, 100000, "50 km", 2, 1)
  north.arrow(49646.41, 2534913, 10000, col = "lightblue");
  title("台灣空氣測站分佈");
  
  par(family = "BiauKai");
  par(mar = c(4, 2, 2, 2));
  boxplot(EPA.STN.attr$PSI[PSI > pr & SiteType != "背景測站"] ~ EPA.STN.attr$SiteType[PSI > pr & SiteType != "背景測站"], data = EPA.STN.attr, drop = TRUE, xlab = "Site Type", ylab = "PSI");
  
  return(pr);
}
