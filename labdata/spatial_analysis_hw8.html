K(d)

density <- Schools.ppp$n / area.default(Schools.ppp$window);

K <- c();
for (i in 1:15) {
  events <- 0;
  for (j in 1:Schools.ppp$n) {
    events <- poly.counts(Schools, gBuffer(Schools[j, ], width = i*1000)) - 1 + events;
  }
  K[i] <- events / (Schools.ppp$n * density);
}
par(mfrow = c(1, 2));
plot(K, main = NULL, xlab = "Distance(km)", ylab = "K", type = "l", lwd = 2);

K.list <- list();
for (k in 1:99) {
  random.K <- c();
  random.points <- as.SpatialPoints.ppp(rpoint(Schools.ppp$n, win = Schools.ppp$window));
  for (l in 1:15) {
    events <- 0;
    for (m in 1:Schools.ppp$n) {
      events <- poly.counts(random.points, gBuffer(random.points[m, ], width = l * 1000)) - 1 + events;
    }
    random.K[l] <- events / (Schools.ppp$n * density);
  }
  lines(random.K, col = rgb(0.5, 0.5, 0.5, 0.1));
  K.list[[k]] <- random.K;
}

plot(envelope(Schools.ppp, fun = Kest), main = "CI");
