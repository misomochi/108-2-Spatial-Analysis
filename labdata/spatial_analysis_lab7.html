nnd.list <- list();
for (i in 1:99) {
  nnd <- nncross(X = rpoint(100, win = Windows.TN), Y = rpoint(nrow(Schools), win = Windows.TN));
  nnd.mean[i] <- mean(nnd$dist);
  nnd.list[[i]] <- nnd;

  plot(F, add = TRUE, col = rgb(0.5, 0.5, 0.5, 0.2), cex = 0, verticals = TRUE, xaxs = "i", yaxs = "i");
}
